三、实验原理
openvela基于NuttX实时操作系统，适用于AIoT嵌入式场景。与普通PC程序不同，本实验需要在Ubuntu中完成面向开发板ARM平台的交叉编译，并将编译打包得到的镜像烧录到开发板后运行
在openvela中，任务可以理解为资源的管理者，线程是系统调度的基本单位。同一任务中的多个线程可以共享全局变量等资源。本实验通过全局变量g_a观察多个线程之间的数据共享和运行先后关系
用户线程通过POSIX线程接口pthread_create创建，其主要参数包括保存线程ID的thread、线程属性attr、线程入口函数startroutine以及传递给线程函数的参数arg。本次实验在hello_main.c中创建add_thread和print_thread两个线程，并结合主线程输出观察线程调度效果
openvela对不同优先级线程采用优先级调度；对相同优先级线程支持FIFO和Round Robin两种调度方式。FIFO为系统默认方式，同优先级线程按创建顺序执行，当前线程需要主动阻塞或让出CPU后才会切换；RR通过CONFIG_RR_INTERVAL配置时间片，当该值大于0时，同优先级线程按时间片轮转运行

(五) FIFO调度机制体验
1. 根据学习手册3.2.3，找到“source/3-2-3_FIFO调度机制体验”中的源代码：
3.2.3部分的上机实验用于观察openvela默认FIFO调度机制下，同优先级线程不主动休眠时的运行现象。FIFO版本代码的关键变化是：主线程、add_thread线程和print_thread线程都去掉sleep休眠，改为较长的空循环延时；同时将线程优先级设置为较高值。由于CONFIG_RR_INTERVAL默认为0，同优先级线程默认采用FIFO调度！完整代码如下：
#include <nuttx/config.h>
#include <stdio.h>
#include <pthread.h>
#include <nuttx/pthread.h>
 static volatile int g_a;
 void * add_thread(void *arg)
 {
  volatile int *p = (volatile int * )arg;
  while (1)
  {
    (*p)++;
    //sleep(1);
    for (volatile int i = 0; i < 100000000; i++);
  }
  return NULL;
 }
void * print_thread(void *arg)
 {
  volatile int *p = (volatile int * )arg;
  while (1)
  {
    printf("val = %d\n", *p);
    //sleep(1);
    for (volatile int i = 0; i < 100000000; i++);
  }
  return NULL;
 }
int main(int argc, FAR char *argv[])
{
  struct sched_param param;
  pthread_t tid1, tid2;
  pthread_attr_t default_attr = { \
    225, /* priority */ \
    PTHREAD_DEFAULT_POLICY,   /* policy */ \
    PTHREAD_EXPLICIT_SCHED,   /* inheritsched */ \
    PTHREAD_CREATE_JOINABLE,  /* detachstate */ \
    0,                        /* affinity */ \
    NULL,                     /* stackaddr */ \
    PTHREAD_STACK_DEFAULT,    /* stacksize */ \
  };
  /* Set current thread priority to 225 */
  param.sched_priority = 225;
  sched_setparam(0, &param);
  pthread_create(&tid1, &default_attr, add_thread, &g_a);  // 100
  pthread_create(&tid2, &default_attr, print_thread, &g_a);
  while (1)
  {
    printf("Hello, World!! g_a = %d\n", g_a);
    //sleep(5);
    for (volatile int i = 0; i < 100000000; i++);
  }
  return 0;
}

(六) RR调度机制体验
1. 根据学习手册3.2.4，将“source/3-2-4_RR调度机制体验”中的源代码替换到apps/examples/hello/hello_main.c里头。完整代码如下：
#include <nuttx/config.h>
#include <stdio.h>
#include <pthread.h>
#include <nuttx/pthread.h>
#include <sched.h>
static void *add_thread(void *arg)
{
  volatile int *p = (volatile int *)arg;
  while (1)
  {
    (*p)++;
    for (volatile int i = 0; i < 100000000; i++);
    sched_yield();
  }
  return NULL;
}
static void *print_thread(void *arg)
{
  volatile int *p = (volatile int *)arg;
  while (1)
  {
    printf("val = %d\n", *p);
    for (volatile int i = 0; i < 100000000; i++);
    sched_yield();
  }
  return NULL;
}
int main(int argc, FAR char *argv[])
{
  struct sched_param param;
  pthread_t tid1;
  pthread_t tid2;
  pthread_attr_t default_attr =
  {
    225,                      /* priority */
    SCHED_RR,                 /* policy */
    PTHREAD_EXPLICIT_SCHED,   /* inheritsched */
    PTHREAD_CREATE_JOINABLE,  /* detachstate */
    0,                        /* affinity */
    NULL,                     /* stackaddr */
    PTHREAD_STACK_DEFAULT,    /* stacksize */
  };
  param.sched_priority = 225;
  sched_setscheduler(0, SCHED_RR, &param);
  pthread_create(&tid1, &default_attr, add_thread, (void *)&g_a);
  pthread_create(&tid2, &default_attr, print_thread, (void *)&g_a);
  while (1)
  {
    printf("Hello, World!! g_a = %d\n", g_a);
    for (volatile int i = 0; i < 100000000; i++);
    sched_yield();
  }
  return 0;
}

五、实验总结
初探实验按照学习手册完成了3.2任务与线程上机实验，完整经历了openvela应用程序从配置、修改源码、编译、打包、烧录到串口运行验证的流程。通过Hello示例程序的验证，进一步明确了开发板上运行的是打包后的系统镜像，因此每次修改hello_main.c后都必须重新m、pack并烧录新镜像
在线程实验中，通过pthread_create创建了多个用户线程，并利用全局变量g_a观察线程之间共享数据和并发运行的现象。3.2.2实验中，主线程、add_thread和print_thread可以共同输出结果，说明线程创建和运行正常
在FIFO调度实验中，由于CONFIG_RR_INTERVAL默认为0，同优先级线程采用FIFO机制，线程不主动休眠时会长时间占用CPU，导致部分线程难以获得运行机会，串口交互也会受到影响。在RR调度实验中，将CONFIG_RR_INTERVAL设置为大于0后，同优先级线程按照时间片轮转运行，输出结果更加均衡，能够更直观地体现时间片调度的作用
通过本次实验，我对openvela中任务、线程和调度机制有了更直观的认识，也熟悉了PhoenixSuit烧录、MobaXterm串口调试以及VS Code远程开发的基本流程。实验过程中遇到的环境变量未加载、配置项误选和串口后台日志干扰等问题，也说明嵌入式开发不仅要关注代码本身，还需要注意系统配置、启动脚本和运行环境等等因素！
